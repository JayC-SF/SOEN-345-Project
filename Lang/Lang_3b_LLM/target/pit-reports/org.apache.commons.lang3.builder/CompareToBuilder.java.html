/**

 *
30

 *   boolean field3;

 *   ...

 *   public int compareTo(Object o) {

 *     return new CompareToBuilder()

 *       .appendSuper(super.compareTo(o)

 *       .append(this.field1, myClass.field1)

 *       .append(this.field2, myClass.field2)

 *   }

 * }
